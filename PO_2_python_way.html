k = 2
theta = 1
n = 1000
mean = stats.gamma.mean(k, scale=theta)
sd = stats.gamma.std(k, scale=theta)
data = np.random.gamma(k, theta, n)
x = np.linspace(0, max(data), 1000)
pdf = stats.gamma.pdf(x, k, scale=theta)
sns.histplot(data,bins=30,stat="density",color="darkblue",label="Histogram danych",alpha=0.7,)
sns.set_style("darkgrid")
sns.lineplot(x=x, y=pdf, label="Funkcja gstoci", color="red")
plt.title(f"Histogram i funkcja gstoci rozkadu Gamma z k = {round(k,2)} i $\\theta$ = {round(theta,2)}")
plt.legend()
plt.show()

z_values = []
s = 10000
n = 1000
for _ in range(s):
    sample = np.random.gamma(k, theta, n)
    sample_mean = np.mean(sample)
    z = np.sqrt(n) * (sample_mean - mean) / sd
    z_values.append(z)
z_values = np.array(z_values)
x = np.linspace(-4, 4, 1000)
normal_pdf = stats.norm.pdf(x, 0, 1)
sns.histplot(z_values, bins=30, stat="density", color="darkblue", label="Histogram Zn", alpha=0.7)
sns.set_style("darkgrid")
sns.lineplot(x=x, y=normal_pdf, label="Gsto N(0,1)", color="red")
plt.title("Histogram Zn i gsto N(0,1)")
plt.legend()
plt.show()
